[-] Building (Makefile).
	REF [project] >> ${SWDT_HOME}/sw_dev/cpp/basic/test/simple_make_test

	REF [file] >> build_software_in_linux.txt
	REF [file] >> build_software_using_gcc.txt

	- Configure.
		Command:
			./configure --prefix=${INSTALL_PREFIX}
			./configure --enable-shared
		Options:
			--enable-static
			--enable-thread
			--with-PACKAGE
			--without-PACKAGE
				--with-openmp
			--includedir=DIR
			--libdir=DIR
			--with-cflags=FLAGS
			--with-cxxflags=FLAGS
			--program-prefix=PREFIX
			--program-prefix=SUFFIX
	- Build.
		Command:
			make
			make all
			make -j4
				The "-j4" is assuming that you have a 4-core CPU.
			make CC=gcc CXX=g++ FC=gfortran
				F77, CPP, LD, CFLAGS, CXXFLAGS, FFLAGS, CPPFLAGS, LDFLAGS.
				Can use environment variables.
			make ARCH=WINNT DEBUG=1
			DEBUG=1 make (?)
	- Clean.
		Command:
			make clean
			make clean DEBUG=1
			make distclean
	- Install.
		Command:
			sudo make install
			sudo make PREFIX=${INSTALL_PREFIX} install

[-] Setting.
	- Print text.
		@echo "text"
			@echo "Hello, 'World!'"

		$(error text)
		$(warning text)
		$(info text)
			$(info PATH is $(PATH))
